# rrm-cli Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a CLI knowledge layer (`rrm-cli`) that gives Claude Code agents fast, intent-aware access to RRM Academy's curated content via Bash.

**Architecture:** Node.js ESM CLI backed by SQLite + FTS5. Reads local JSON data files, syncs into a queryable database at `~/.rrm-cli/rrm.db`. Intelligence layer (relationships, annotations, query log) grows with use. Live Vectorize endpoint as optional semantic search.

**Tech Stack:** Node.js (ESM), better-sqlite3, commander, native fetch (Node 18+)

**Spec:** `docs/superpowers/specs/2026-03-15-rrm-cli-design.md`

```js
import fs from 'node:fs';
import path from 'node:path';
import os from 'node:os';

const DEFAULT_CONFIG_DIR = path.join(os.homedir(), '.rrm-cli');
const DEFAULT_CONFIG_PATH = path.join(DEFAULT_CONFIG_DIR, 'config.json');

export function loadConfig(configPath = DEFAULT_CONFIG_PATH) {
  try {
    const raw = fs.readFileSync(configPath, 'utf-8');
    return JSON.parse(raw);
  } catch {
    return null;
  }
}

export function saveConfig(configPath = DEFAULT_CONFIG_PATH, config) {
  const dir = path.dirname(configPath);
  fs.mkdirSync(dir, { recursive: true });
  fs.writeFileSync(configPath, JSON.stringify(config, null, 2) + '\n');
}

export function resolveDbPath(dbPath) {
  if (dbPath.startsWith('~')) {
    return path.join(os.homedir(), dbPath.slice(1));
  }
  return dbPath;
}

export function getConfigDir() {
  return DEFAULT_CONFIG_DIR;
}

export function getDefaultConfig() {
  return {
    dataDir: '',
    siteUrl: 'https://rrmacademy.org',
    dbPath: path.join(DEFAULT_CONFIG_DIR, 'rrm.db'),
    queryLogging: true
  };
}
```

```js
    assert.ok(tables.includes('relationships'));
```

```js
import Database from 'better-sqlite3';
import fs from 'node:fs';
import path from 'node:path';

export function openDb(dbPath, { readOnly = false } = {}) {
  const dir = path.dirname(dbPath);
  if (!readOnly) {
    fs.mkdirSync(dir, { recursive: true });
  }
  const db = new Database(dbPath, { readonly: readOnly });
  db.pragma('journal_mode = WAL');
  db.pragma('foreign_keys = ON');
  return db;
}

export function hasContent(db) {
  try {
    const row = db.prepare('SELECT COUNT(*) as count FROM content').get();
    return row.count > 0;
  } catch {
    return false;
  }
}
```

```js
export class CliError extends Error {
  constructor(code, message) {
    super(message);
    this.code = code;
  }
}

export function handleError(err, pretty = false) {
  const payload = {
    error: err.code || 'unknown',
    message: err.message
  };
  if (pretty) {
    process.stderr.write(`Error: ${payload.message}\n`);
  } else {
    process.stderr.write(JSON.stringify(payload) + '\n');
  }
  process.exit(1);
}
```
